class Solution {
    vector<int> mmp[16];
    set<int> s;
    bool book[16];
    int ans;
public:
    void init()
    {
        for (int i = 0; i < 16; i++)mmp[i].clear();
        for (int i = 0; i < 16; i++)book[i] = true;
        ans = 0;
    }
    void dfs(int cnt, int end)
    {
        //printf();
        if (cnt == end){ans++; return;}   
        for (int i = 0; i < mmp[cnt].size(); i++)
        {
            if (book[mmp[cnt][i]])
            {
                //printf("%d", mmp[cnt][i]);
                book[mmp[cnt][i]] = false;
                dfs(cnt+1, end);
                book[mmp[cnt][i]] = true;
            }
            
        }
    }
    int countArrangement(int n) {
        init();
        for (int i = 1; i <= n; i++)
        {
            for (int j = 1; j <= n; j++)
                if ((i%j == 0)||(j%i == 0))
                    mmp[i].push_back(j);
        }
        dfs(1, n+1);
        return ans;
    }
};
